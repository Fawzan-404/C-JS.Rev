1. Variables and Data Types

Variables

let name = "Tony"; // Reassignable
const pi = 3.14; // Constant, cannot be reassigned
var age = 25; // Function-scoped variable

Data Types

let number = 100;       // Number
let name = "Alice";     // String
let isActive = true;   // Boolean
let value = null;      // Null
let notDefined;        // Undefined
let uniqueSymbol = Symbol('id'); // Symbol
let bigIntNumber = 1234567890123456789012345678901234567890n; // BigInt

2. Operators

Arithmetic Operators

let sum = 5 + 3; // Addition
let diff = 10 - 4; // Subtraction
let product = 2 * 3; // Multiplication
let quotient = 8 / 2; // Division
let remainder = 7 % 3; // Modulus

Comparison Operators

let isEqual = (5 == 5); // Equal to
let isNotEqual = (5 != 3); // Not equal to
let isGreaterThan = (10 > 5); // Greater than
let isLessThan = (3 < 7); // Less than
let isGreaterOrEqual = (5 >= 5); // Greater than or equal to
let isLessOrEqual = (3 <= 4); // Less than or equal to

3. Control Flow

Conditional Statements

if (age > 18) {
    console.log("Adult");
} else {
    console.log("Minor");
}

switch (day) {
    case 1:
        console.log("Monday");
        break;
    case 2:
        console.log("Tuesday");
        break;
    default:
        console.log("Other day");
}

4. Functions

Function Declaration

function greet(name) {
    return "Hello, " + name;
}

console.log(greet("Alice")); // Hello, Alice

Arrow Functions

const add = (a, b) => a + b;
console.log(add(5, 3)); // 8

5. Objects and Arrays

Objects

let person = {
    name: "John",
    age: 30,
    greet: function() { return "Hello!"; }
};

console.log(person.name); // John
console.log(person.greet()); // Hello!

Arrays

let numbers = [1, 2, 3, 4, 5];
console.log(numbers[0]); // 1
numbers.push(6); // Adding an element
console.log(numbers.length); // 6

element.innerHTML = "New Content";

Event Listeners

8. Asynchronous JavaScript

Callbacks

function fetchData(callback) {
    setTimeout(() => {
        callback("Data received");
    }, 1000);
}

fetchData((message) => {
    console.log(message);
});

Promises

let promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve("Data received");
    }, 1000);
});

promise.then((message) => {
    console.log(message);
});

Async/Await

async function fetchData() {
    let response = await new Promise((resolve) => {
        setTimeout(() => {
            resolve("Data received");
        }, 1000);
    });
    console.log(response);
}

fetchData();
